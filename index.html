// إعداد API على الموقع لاستقبال البيانات من الذكاء الصناعي const express = require('express'); const cors = require('cors'); const app = express(); const PORT = 3000; app.use(express.json()); app.use(cors()); // نقطة نهاية لاستقبال بيانات الذكاء الصناعي app.post('/api/update-data', (req, res) => { const receivedData = req.body; console.log('بيانات مستلمة:', receivedData); // تحديث البيانات في الموقع أو قاعدة البيانات // سيتم إضافة الكود الخاص بالذكاء الصناعي لاحقًا res.json({ message: 'تم استقبال البيانات بنجاح', status: 'success' }); }); app.listen(PORT, () => { console.log(`🚀 الخادم يعمل على http://localhost:${PORT}`); });
